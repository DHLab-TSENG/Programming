url="https://datacenter.taichung.gov.tw/swagger/OpenData/34a848ab-eeb3-44fd-a842-a09cb3209a7d"

data = response.json()  # Convert the response to JSON

path = 'output_writeline.txt'
f = open(path, 'w')
lines1 = ['Hello World', '123', '456', '789']

f.writelines(lines1)
f.writelines(lines2)
f.close()
